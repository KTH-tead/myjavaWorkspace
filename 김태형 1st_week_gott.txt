3. for문을 이용해서 1부터 100까지 정수 중에서 3의 배수의 총합을 구하는 코드를 작성해 보세요.
    public static void main(String[] args) {
		int sum = 0;
		int i;
		for ( i=3; i<100; i+=3) {
			sum += i;
		}
		System.out.println(sum);
	}
